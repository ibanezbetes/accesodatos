-- EN LA FUNCIÓN LE PASO EL APELLIDO QUE SERÁ UN TIPO DE DATO DINAMICO, COGERÁ
-- EL TIPO EN TIEMPO DE EJECUCIÓN DE LA TABLA EMPLE DEL CAMPO apellido.
-- POR DEFECTO TOMARÁ EL VALOR DE 'GIL', ESTO QUIERE DECIR QUE LA FUNCIÓN 
-- SE PODRÁ EJECUTAR DE DOS MANERAS, PASANDOLE UN APELLIDO O SIN APELLIDO, EN ESE
-- CASO COGERÉ EL APILLEDO 'GIL'.
-- LA FUNCIÓN DEVUELVE UN NÚMERO QUE ES TIPO NUMBER PORQUE SERÁ EL Nº DE EMPLEADO.

CREATE FUNCTION Encontrar_emple(v_apellido EMPLE.apellido%TYPE DEFAULT 'GIL')
RETURN NUMBER
AS
/* Zona de declaración de variables*/
v_numemple EMPLE.emp_no%TYPE;

BEGIN
/* Zona de código*/
   SELECT emp_no INTO v_numemple FROM EMPLE
   WHERE UPPER(apellido) = UPPER(v_apellido);
/* Comparación en Mayus. por problemas de ASCIIs 'a' != 'A'
 P.e. 'López' != 'LÓPEZ' para SQL*/
   
   RETURN v_numemple;/* Aqui solo llego sino hay errores, por ejemplo existe el empleado */

/* Fin del prog. ppal (de la función)*/

EXCEPTION
/* Código de manejo de errores  ver pg 290
// Error 1: No hay permisos del usuario sobre EMPLE
// Error 2: Que haya más de un empleado con ese apellido
// Error 3: Que no haya ningún empleado con ese apellido
*/
WHEN TOO_MANY_ROWS THEN  /* Error 2 */
   DBMS_OUTPUT.PUT_LINE('Error !!! Hay más de un empleado con el apellido ' || v_apellido);
   RETURN(-2);

WHEN NO_DATA_FOUND THEN  /* Error 3 */
   DBMS_OUTPUT.PUT_LINE('Error !!! No hay ningún empleado con el apellido ' || v_apellido);
   RETURN(-3);
   
WHEN OTHERS THEN /* Error 1 & otros */
   DBMS_OUTPUT.PUT_LINE(' Error desconocido !!!');
   RETURN(-1);

END Encontrar_emple;
/
