DROP PROCEDURE BORRAR_DPTO
SET SERVEROUTPUT ON

CREATE PROCEDURE BORRAR_DPTO
AS
  v_dept99 NUMBER(2);
  v_num_empleados NUMBER(2);

BEGIN
  SELECT COUNT(*) INTO v_dept99 FROM DEPART WHERE dept_no =99;  

  IF (v_dept99 = 0)
   THEN
     INSERT INTO DEPART VALUES (99, 'TEMPORAL', 'TEMPORAL');
  END IF;

  UPDATE EMPLE SET dept_no = 99 WHERE dept_no = 10;
  v_num_empleados := SQL%ROWCOUNT;
  DBMS_OUTPUT.PUT_LINE(v_num_empleados || ' empleados ubicados en TEMPORAL  ');
  DELETE FROM DEPART WHERE dept_no = 10;
END;
/

  v_num_empleados NUMBER(2);

BEGIN
  SELECT COUNT(*) INTO v_dept99 FROM DEPART WHERE dept_no =99;  

  IF (v_dept99 = 0)
   THEN
     INSERT INTO DEPART VALUES (99, 'TEMPORAL', 'TEMPORAL');
  END IF;

  UPDATE EMPLE SET dept_no = 99 WHERE dept_no = 

p_dept_no   ;
  v_num_empleados := SQL%ROWCOUNT;
  DBMS_OUTPUT.PUT_LINE(v_num_empleados || ' empleados ubicados en TEMPORAL  ');
  DELETE FROM DEPART WHERE dept_no = 

p_dept_no   ;
END;
