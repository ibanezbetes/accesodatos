CREATE PROCEDURE modificar_precio_producto
          (numproducto NUMBER, nuevoprecio NUMBER)
AS
	v_precioant PRODUCTOS.precio%TYPE;
BEGIN
   SELECT precio INTO v_precioant FROM PRODUCTOS
   WHERE cod_producto = numproducto; 
   DBMS_OUTPUT.PUT_LINE('El producto existe !!!');
   
   IF (v_precioant *0.20)>(nuevoprecio - v_precioant) THEN
      UPDATE PRODUCTOS SET precio = nuevoprecio
        WHERE cod_producto = numproducto;
      DBMS_OUTPUT.PUT_LINE('Modificación realizada con éxito !!!');
   ELSE
      DBMS_OUTPUT.PUT_LINE('Error !!!  La modificación supera el 20% del precio original !!!');
   END IF;
   
   
EXCEPTION
   WHEN NO_DATA_FOUND THEN
      DBMS_OUTPUT.PUT_LINE('No existe el producto '||numproducto);
      
END;
/

SET SERVEROUTPUT ON
EXECUTE modificar_precio_producto(60, 100);

EXECUTE modificar_precio_producto(13, 14);

EXECUTE modificar_precio_producto(13, 8);
